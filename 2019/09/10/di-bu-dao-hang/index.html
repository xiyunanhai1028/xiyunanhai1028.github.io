class BottomNavigationBarWidget extends StatefulWidget {
  _BottomNavigationBarWidgetState createState() =>
      _BottomNavigationBarWidgetState();
}

class _BottomNavigationBarWidgetState extends State<BottomNavigationBarWidget> {
  int _currentIndex = 0;
  Color actionColor = Colors.blue;
  List<Widget> _lists = List();

  @override
  void initState() {
    _lists..add(HomePage())..add(SearchPage())..add(MePage());
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: _lists[_currentIndex],
      bottomNavigationBar: BottomNavigationBar(
        type: BottomNavigationBarType.fixed,
        onTap: (int index) {
          this.setState(() {
            _currentIndex = index;
          });
        },
        currentIndex: _currentIndex,
        items: [
          BottomNavigationBarItem(
              icon: Icon(Icons.home),
              title: Text("首页"),
              activeIcon: Icon(Icons.home, color: actionColor)),
          BottomNavigationBarItem(
              icon: Icon(Icons.search),
              title: Text("搜索"),
              activeIcon: Icon(Icons.search, color: actionColor)),
          BottomNavigationBarItem(
              icon: Icon(Icons.person),
              title: Text("我的"),
              activeIcon: Icon(Icons.person, color: actionColor))
        ],
      ),
    );
  }
}

void main() => runApp(App());

class App extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: TabWidget(),
    );
  }
}

import 'package:flutter/material.dart';

class TabWidget extends StatefulWidget {
  _TabWidgetState createState() => _TabWidgetState();
}

class _TabWidgetState extends State<TabWidget>
    with SingleTickerProviderStateMixin {
  TabController _controller;

  @override
  void initState() {
    _controller = TabController(length: 3, vsync: this);
  }

  @override
  void dispose() {
    _controller.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("tabBar"),
        centerTitle: true,
        bottom: TabBar(controller: _controller, tabs: [
          Tab(icon: Icon(Icons.accessible)),
          Tab(icon: Icon(Icons.airplanemode_active)),
          Tab(icon: Icon(Icons.motorcycle)),
        ]),
      ),
      body: TabBarView(controller: _controller, children: [
        Center(
          child: Text("111",style: TextStyle(fontSize: 30,color: Colors.redAccent),),
        ),
        Center(
          child: Text("222",style: TextStyle(fontSize: 30,color: Colors.greenAccent)),
        ),
        Center(
          child: Text("333",style: TextStyle(fontSize: 30,color: Colors.blue)),
        )
      ]),
    );
  }
}

List<String> list = ["张三", "张可欣", "张无忌", "张翠山", "宋远桥", "宋青书", "王鸥", "王健林"];

const defaultData = ["帅哥", "靓妹"];
import 'package:flutter/material.dart';
import 'widget/search_bar_widget.dart';

void main() => runApp(App());

class App extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: SearchBarWidget(),
    );
  }
}
import 'package:flutter/material.dart';
import '../data/search_data.dart';

class SearchBarWidget extends StatefulWidget {
  _SearchBarWidgetState createState() => _SearchBarWidgetState();
}

class _SearchBarWidgetState extends State {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("search"),
        centerTitle: true,
        actions: <Widget>[
          IconButton(
              icon: Icon(Icons.search),
              onPressed: () {
                list = ["张三", "张可欣", "张无忌", "张翠山", "宋远桥", "宋青书", "王鸥", "王健林"];
                showSearch(context: context, delegate: _SearchBarDelegate());
              })
        ],
      ),
      body: ListView(
        children: _ListItems(),
      ),
    );
  }

  List<Widget> _ListItems() {
    List<Widget> items = List();
    list.forEach((item) => {
          items.add(ListTile(
            title: Text(item),
          ))
        });
    return items;
  }
}

class _SearchBarDelegate extends SearchDelegate<String> {
  @override
  List<Widget> buildActions(BuildContext context) {
    return [
      IconButton(
          icon: Icon(Icons.clear),
          onPressed: () {
            query = "";
            close(context, null);
          })
    ];
  }

  @override
  Widget buildLeading(BuildContext context) {
    return IconButton(
        icon: Icon(Icons.arrow_back),
        onPressed: () {
          close(context, null);
        });
  }

  @override
  Widget buildResults(BuildContext context) {
    close(context, null);
    return ListTile(
      title: Text(query),
    );
  }

  @override
  Widget buildSuggestions(BuildContext context) {
    final data = query.isEmpty
        ? defaultData
        : list.where((item) => (item.startsWith(query))).toList();
    return ListView(
      children: _ListItems(data, context),
    );
  }

  List<Widget> _ListItems(List<String> data, BuildContext context) {
    List<Widget> items = List();
    data.forEach((item) => {
          items.add(ListTile(
            onTap: () {
              print(item);
              query = item;
              list = list.where((t) => (t.startsWith(item))).toList();
              close(context, null);
            },
            title: RichText(
                text: TextSpan(
                    text: item.substring(0, query.length),
                    style: TextStyle(
                        color: Colors.black, fontWeight: FontWeight.bold),
                    children: [
                  TextSpan(
                      text: item.substring(query.length),
                      style: TextStyle(color: Colors.grey))
                ])),
          ))
        });
    return items;
  }
}

import 'package:flutter/material.dart';

class ClipperWidget extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("贝瑟尔曲线一"),
        centerTitle: true,
      ),
      body: ClipPath(
        clipper: BottomClipper(),
        child: Container(height: 200, color: Colors.blue),
      ),
    );
  }
}

class BottomClipper extends CustomClipper<Path> {
  @override
  Path getClip(Size size) {
    var path = Path();
    path.lineTo(0, 0);
    path.lineTo(0, size.height - 50);
    var firstControlPath = Offset(size.width / 4, size.height);
    var firstEndPath = Offset(size.width/2, size.height - 40);
    path.quadraticBezierTo(firstControlPath.dx, firstControlPath.dy,
        firstEndPath.dx, firstEndPath.dy);

    var secondControlPath = Offset(size.width / 4*3, size.height-70);
    var secondEndPath = Offset(size.width, size.height - 65);
    path.quadraticBezierTo(secondControlPath.dx, secondControlPath.dy,
        secondEndPath.dx, secondEndPath.dy);
    path.lineTo(size.width, size.height - 65);
    path.lineTo(size.width, 0);
    return path;
  }

  @override
  bool shouldReclip(CustomClipper<Path> oldClipper) {
    // TODO: implement shouldReclip
    return false;
  }
}
