1. babel安装：

  • 全局安装：npm install -g babel-cli

  • 本地安装：npm install –save-dev babel-preset-es2015 babel-cli

  • 创建.babelrc文件

    • {
  "presets": [
    "es2015"
  ],
  "plugins": []
}

  • 转换: babel src/index.js -o dist/index.js

    • -o：转换
    • 意思：将src下的index.js文件转化为dis文件下的index.js文件

  • 配置package.json 

    • "scripts": {
    "build":"babel src/index.js -o dist/index.js"
  },

  • npm run build

2. var,let,const区别

var:全局变量，容易被污染
let:局部变量
const：常量

3. null,undefined区别

null:有值，但是为空

undefined：没有值，未定义

7. indexOf ,includes, startsWith, endtsWith, repeat

let a='23'
let b='123456789'
console.log(b.indexOf(a))//1
console.log(b.includes(a))//true
console.log(b.startsWith(a))//false
console.log(b.endsWith(a))//false
console.log(a.repeat(2))//2323

8. ES6数字操作

Number.isFinite():是否位数字
Number.isNaN():是否不是数字
Number.isInteger():是否是整数
Number.parseInt():转化为整数
Number.parseFloat():转为浮点型
Number.MAX_SAFE_INTEGER：最大安全数
Number.MIN_SAFE_INTEGER：最小安全数
Number.isSafeInteger():安全整数判断

9. ES6中新增的数组知识

9.1 Array.of()

 const arr=Array.of(1,2,3,4)
    console.log(arr)

9.2 find()

const a=[1,2,3,4,5,6,7]
const b=a.find((val)=>val===5)
console.log(b)

9.3 fill()

    let a=['1','2','3']
    a.fill('1',1,2)
    console.log(a)

9.4 for of

    let a=['1','2','3']
    for( let item of a){
        console.log(item)
    }
    for(let key of a.keys()){
        console.log(key)
    }
    for(let item of a.entries()){
        console.log(item)
    }
    for(let [index,item] of a.entries()){
        console.log(index)
        console.log(item)
    }

9.5 Array.from()

json转数组，必须有length

const json={
    "0":"张三",
    "1":"男",
    "2":18,
    length:3
}
let arr=Array.from(json)
console.log(arr)

13. ES6中对象

13.1 对象赋值

    const a="jack"
    const b='tom'
    const c={a,b}
    console.log(c)//{a: "jack", b: "tom"}

13.2 对象Key值构建

    let a='name'
    const b={
        [a]:'zhangsan'
    }
    console.log(b)//{name: "zhangsan"}

13.3 Object.is() 对象比较

===:同值相等 is:严格相等

    console.log(+0===-0)//true
    console.log(NaN===NaN)//false
    console.log(Object.is(+0,-0))//false
    console.log(Object.is(NaN,NaN))//true

13.4 Object.assign( )合并对象

    const a={"name":"张三"}
    const b={"sex":"男"}
    console.log(Object.assign(a,b))//{name: "张三", sex: "男"}

16. Proxy:预处理

let pro = new Proxy({
    'name': '张三',
    'age': 18
}, {
    get(target, key, property){
        return target[key]
    },
    set(target, key, value, receiver){
        return target[key] = value
    },
    apply(target, ctx, args){
        return Reflect.apply(...arguments)
    }
})
pro.name='Jack'
console.log(pro.name) //Jack

17. promise对象的使用

es6中的promise的出现给我们很好的解决了回调地狱的问题

let step1 = (resolve, reject) => resolve("打开冰箱门")
let step2 = (resolve, reject) => resolve("把大象放冰箱")
let step3 = (resolve, reject) => resolve("关冰箱门")
new Promise(step1).then((val) => {
        console.log(val)
        return new Promise(step2)
    }
).then(
    (val) => {
        console.log(val)
        return new Promise(step3)
    }
).then(
    (val) => {
        console.log(val)
        return val
    }
)
